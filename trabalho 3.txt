#include <iostream>
#include <vector>
#include <math.h>

using namespace std;

double quadratic(double var, vector<double> param){

    return var*var*param[0]+var*param[1]+param[2];

}
double linear(double var, vector<double> param)
{
    return var*param[0]+param[1];
}

double Riemman(double (*func)(double,vector<double>),vector<double> param,double interi,double interf,int retangulos){

    double area = 0;
    double variacao = fabs((interf-interi))/retangulos,inicial = interi + variacao;

    for (double i = inicial;i<interf; i+= variacao)
    {
        area += variacao * func(i,param);
    }

    return area;

}
