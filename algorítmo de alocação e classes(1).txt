#include <iostream>

using namespace std;

class matriz{


    public:
        matriz(int a,int b){
        mxn = new int*[a];
        for(i=0;i<a;i++){
            mxn[i]=new int[b];
        }
        h=&a;
        }

        void valor_matriz(int a,int b)
        {
            int j,k;
            for(j=0;j<a;j++)
            {
                for(k=0;k<b;k++){
                    mxn[j][k] = j*k;
                    cout << mxn[j][k] << " ";
                }
                cout << "\n";
            }

        }

        int *operator[](int l)
        {
            return mxn[l];
        }

        ~matriz()
        {
            for(i=0;i<*h;i++){
                delete mxn[i];
            }
        }

    private:
        int **mxn,i,*h;

};


int main()
